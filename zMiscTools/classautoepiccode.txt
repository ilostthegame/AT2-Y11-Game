# Attributes
__surf = None
__rect = None

# Constructor
def __init__(self, surf, rect):
    self.setSurf(surf)
    self.setRect(rect)

# Getters
def getSurf(self):
    return self.__surf
def getRect(self):
    return self.__rect

# Setters
def setSurf(self, surf):
    self.__surf = surf
def setRect(self, rect):
    self.__rect = rect
